\section{\makebox[\widthof{\faUsers}][c]{\color{CVBlue}\faUsers}\ 
科研经历}

\textbf{UNOP：面向不规则3D几何的
长时序世界模型} 
\textit{(一作，IJCAI CCF-A 审稿中)} 
\hfill 2025.3 -- 至今
\begin{itemize}
    \item \textbf{问题定义：}
    在复杂3D几何（非结构化网格）上
    进行\textbf{长序列自回归生成}，
    核心挑战与世界模型一致：
    \textbf{(1)} 长时序自回归漂移与
    发散（现有方法$\sim$20步崩溃）；
    \textbf{(2)} 不规则3D拓扑下的
    空间一致性维护；
    \textbf{(3)} 训练数据获取成本极高。
    
    \item \textbf{3D几何编码：}
    设计\textbf{几何解耦编码器(GALA)}，
    将不规则3D网格分解为局部几何
    基函数，统一异构拓扑的表征——
    类似3DGS中用基元表达
    复杂3D场景的思路。
    
    \item \textbf{长序列一致性：}
    提出基于\textbf{积分一致性}的
    全局约束(LIPE)，替代逐点微分
    监督，从\textbf{全局守恒律}角度
    约束自回归生成的一致性，
    3D内存降至微分方法的
    \textbf{1/6}。
    
    \item \textbf{结构感知生成：}
    设计\textbf{曲率感知门控(GSEO)}，
    在自回归rollout中自适应
    增强高梯度区域（边缘/
    突变/细节结构）的生成精度。
    
    \item \textbf{结果：}3D ShapeNet-Car 
    误差\textbf{4.06\%}，
    \textbf{50步自回归外推仍稳定}
    （SOTA $\sim$20步发散），
    \textbf{无需任何标注数据}训练。
\end{itemize}
