% --- 科研一：UNOP ---
\textbf{UNOP：无监督神经算子用于长时序预测} 
\textit{(一作，IJCAI CCF-A 审稿中)} 
\hfill 2025.3 -- 至今
\begin{itemize}
    \item 针对长时序预测中\textbf{误差累积与稳定性不足}问题，设计无监督神经算子框架，摆脱对仿真标签的依赖。
    
    \item 提出基于积分一致性的物理约束（LIPE），\textbf{替代高阶自动微分}，将3D训练内存开销降低至\textbf{1/6}。
    
    \item 设计几何解耦编码器（GALA）与曲率感知门控机制（GSEO），提升非结构化几何数据下的表示能力与稳定性。
    
    \item \textbf{实验结果：}1D误差\textbf{0.09\%}（优于PINO 36$\times$），2D \textbf{3.02\%}（优于SP-FNO 4.5$\times$），3D Cd误差\textbf{4.06\%}，\textbf{50步外推稳定}（传统方法约20步发散）。
\end{itemize}

\textbf{GeoGatedAttention：几何感知的稀疏图Transformer} 
\textit{(发明专利已受理)} \hfill 2024.9 -- 2025.3
\begin{itemize}
    \item 针对复杂结构数据中\textbf{关键区域建模不足}问题，设计几何感知注意力机制。
    \item 引入边级几何特征作为门控信号，动态调制attention权重，实现特征相关性与几何重要性的融合。
    \item 基于图稀疏结构实现\textbf{O(E)}复杂度，在保证表达能力的同时提升计算效率。
    \item \textbf{结果：}压力场误差下降\textbf{40\%}，Cd误差\textbf{2.69\%}，推理时间\textbf{20ms}。
\end{itemize}
