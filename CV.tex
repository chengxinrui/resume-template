% --- 科研一：UNOP ---
\textbf{UNOP：面向物理世界动态建模的无监督生成式预测框架} 
\textit{(一作，IJCAI CCF-A 审稿中)} 
\hfill 2025.3 -- 至今
\begin{itemize}
    \item 设计\textbf{无监督世界模型框架}，在无仿真标签条件下自回归生成高维物理状态序列，解决长时序生成中\textbf{误差累积与时序一致性}问题。
    \item 提出积分一致性物理约束（LIPE）作为\textbf{新型目标函数}，替代高阶自动微分；3D场景训练\textbf{显存降低至1/6}，使大规模3D世界模型训练成为可能。
    \item 设计几何解耦编码器（GALA）+ 曲率感知门控（GSEO），增强模型对\textbf{非结构化3D几何}的表示与生成能力，支持任意网格拓扑。
    \item \textbf{实验结果：}1D误差\textbf{0.09\%}（优于PINO 36$\times$），2D \textbf{3.02\%}（优于SP-FNO 4.5$\times$），3D Cd误差\textbf{4.06\%}，\textbf{50步外推稳定}（传统方法约20步发散）。
\end{itemize}

\textbf{GeoGatedAttention：面向3D几何数据的稀疏图Transformer} 
\textit{(发明专利已受理)} \hfill 2024.9 -- 2025.3
\begin{itemize}
    \item 针对3D非结构化数据中\textbf{关键区域建模不足}问题，设计几何感知\textbf{Transformer注意力机制}，可作为3D生成模型中的通用注意力模块。
    \item 引入边级几何特征作为\textbf{门控条件信号}动态调制attention权重，实现几何条件与特征注意力的深度融合。
    \item 基于图稀疏结构实现\textbf{O(E)}复杂度，兼顾大规模3D数据的表达能力与计算效率。
    \item \textbf{结果：}压力场误差下降\textbf{40\%}，Cd误差\textbf{2.69\%}，单帧推理\textbf{20ms}。
\end{itemize}
