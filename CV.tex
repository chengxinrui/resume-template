\section{\makebox[\widthof{\faUsers}][c]{\color{CVBlue}\faUsers}\ 科研经历}

% --- 项目一（契合自动驾驶时空预测与物理演化底层框架） ---
\textbf{UNOP：面向复杂物理场景的 3D 时空推演与预测架构} \textit{(核心开发者/一作，IJCAI CCF-A 审稿中)} \hfill 2025.9 -- 至今
\begin{itemize}
    \item \textbf{项目背景：}针对长时序环境状态预测中极易产生自回归误差发散、违背物理规律的痛点，设计了具备高保真物理一致性的 3D 时空连续演化底层架构。
    \item \textbf{长时序预测稳定性 (LIPE)：}摒弃极易产生数值爆炸的微分残差，创新性引入基于 Feynman-Kac 的积分物理约束。在无监督环境下，大幅提升了极端长视距 (Long-horizon) 连续物理态推演的稳定性，缓解了时序预测的误差累积问题。
    \item \textbf{3D 点云空间表征 (GALA)：}结合图注意力网络与空间锚点投影，将非结构化的复杂 3D 点云（$\sim$5000点规模）高效映射至统一的规则三维网格（Latent Grid）。突破了传统模型难以直接处理复杂 3D 空间拓扑的瓶颈。
    \item \textbf{动态演化控制算子 (GSEO)：}结合频域演化与曲率感知门控机制，自适应保留时空预测过程中的高频剧烈变化区域，有效防止了在连续多帧演化预测后出现的模糊与物理失真。
    \item \textbf{核心成果：}在 1D/2D 连续时空推演任务中，长程预测误差低至 \textbf{0.09\%} 和 \textbf{3.02\%}；在复杂的 3D 表面流场空间演化中，核心物理指标误差仅 \textbf{4.06\%}。
\end{itemize}

% --- 项目二（契合自动驾驶/机器人底层仿真测试引擎建设） ---
\textbf{基于图神经网络的复杂 3D 物理环境高频仿真与预警系统} \textit{(AI算法核心开发者)} \hfill 2024.9 -- 2025.6
\begin{itemize}
    \item \textbf{项目背景：}针对复杂机械在传统物理引擎（FEM）下仿真计算耗时长的痛点，开发基于\textbf{神经算子(Neural Operator)}的毫秒级实时 3D 物理场高频仿真预测平台，为环境推演提供加速支持。
    \item \textbf{AI 空间特征建模：}引入 Geo-FNO 构建可微坐标变换以处理非规则 3D 几何；独立设计了几何门控图神经网络 (GNN)，利用表面曲率特征生成动态注意力权重，精准捕捉环境高频形变，使复杂物理态预测误差\textbf{降低 40\% 以上}。
    \item \textbf{虚实多模态数据比对：}实现 3D 拓扑与物理参数的动态解耦及高频未来态推演；将 AI 加速预测的理论时空基线与真实物理传感器回传数据进行实时差分比对，实现物理异常的精准捕获。
    \item \textbf{端到端系统集成：}协助团队打通了“3D物理态仿真预测-传感器状态评估-大模型诊断”的完整闭环，构建了初步的智能物理监控链路。
    \item \textbf{项目成果：}有效缓解了多物理场仿真的算力瓶颈。核心算法已申请\textbf{3项发明专利}及\textbf{2项软著}，并获\textbf{第十五届中兴捧月区域优胜奖}。
\end{itemize}
