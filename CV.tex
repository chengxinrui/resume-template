\section{\makebox[\widthof{\faUsers}][c]{\color{CVBlue}\faUsers}\ 科研经历}

% --- 项目一（最重要，放最前，IJCAI在审） ---
\textbf{UNOP：物理约束无监督神经算子} \textit{(核心开发者/一作，IJCAI CCF-A 审稿中)} \hfill 2025.9 -- 至今
\begin{itemize}
    \item \textbf{项目背景：}针对3D复杂几何神经算子训练中的梯度爆炸与长时发散问题，设计无监督PDE实时求解框架。
    \item \textbf{独立提出 LIPE 积分物理嵌入：}基于Feynman-Kac公式将PDE微分残差转化为蒙特卡洛积分约束，\textbf{消除高阶自动微分显存指数膨胀}，支持高分辨率网格稳定训练。
    \item \textbf{独立设计 GALA 几何适配器：}结合图注意力网络与空间锚点投影，将$\sim$5000点复杂点云映射至统一潜空间网格，实现几何结构与物理演化解耦。
    \item \textbf{独立开发 GSEO 门控谱演化算子：}结合FFT频域卷积与曲率感知门控机制，保持长时PDE rollout激波稳定。
    \item \textbf{核心成果：}1D/2D任务预测误差低至\textbf{0.09\%}和\textbf{3.02\%}，优于PINO和SP-FNO；3D阻力系数误差仅\textbf{4.06\%}。
\end{itemize}

\textbf{水轮机数字孪生与智能运维系统} \textit{(核心开发者)} \hfill 2024.9 -- 2025.6
\begin{itemize}
    \item \textbf{项目背景：}针对水电机组多物理场耦合难题，构建\textbf{毫秒级}实时数字孪生系统，替代传统FEM计算。
    \item \textbf{独立设计几何门控图神经网络：}利用法向、曲率等几何特征生成动态注意力因子，精准捕捉应力集中与流场突变，\textbf{使高曲率区域预测误差降低40\%以上}。
    \item \textbf{独立提出 Geo-FNO 框架：}构建可微坐标变换与几何编码，映射非规则物理空间，\textbf{计算效率提升50\%--80\%}。
    \item \textbf{独立构建 PINO 驱动耦合模型：}融合机理与数据驱动建立四级健康评估体系，实现自动降载与停机闭环控制。
    \item \textbf{核心成果：}单次3D应力场推演仅需\textbf{20ms}，效率提升\textbf{3个数量级}，关键区域误差\textbf{$<$3\%}。
\end{itemize}
