\section{\makebox[\widthof{\faUsers}][c]{\color{CVBlue}\faUsers}\ 科研经历}

% --- 项目一（最重要，高度契合时空生成大模型与AI4S） ---
\textbf{UNOP：基于物理约束的无监督长时序生成架构} \textit{(核心开发者/一作，IJCAI CCF-A 审稿中)} \hfill 2025.9 -- 至今
\begin{itemize}
    \item \textbf{项目背景：}针对现有时空生成大模型在长序列预测中极易产生自回归误差累积、违背物理规律的痛点，设计了具备高保真物理一致性的 3D 时空演化基础框架。
    \item \textbf{长时序稳定演化 (LIPE)：}摒弃极易产生数值爆炸的微分残差，创新性引入基于 Feynman-Kac 的积分物理约束。有效抑制了高频噪声与梯度病态，\textbf{在缺乏真实仿真标签的情况下，大幅提升了极端长视距(Long-horizon rollout)推演的物理一致性与系统稳定性}。
    \item \textbf{3D 空间表征适配器 (GALA)：}结合图注意力网络与空间锚点投影机制，将非结构化的复杂 3D 点云（$\sim$5000点）高效映射至统一的规则潜在网格（Latent Grid），突破了传统大模型难以直接处理非欧几里得 3D 几何的瓶颈。
    \item \textbf{时空动态演化算子 (GSEO)：}结合频域演化与曲率感知门控机制，自适应保留时空预测过程中的高频剧烈变化区域（如边界突变、激波等），防止连续多帧生成后出现模糊失真。
    \item \textbf{核心成果：}在 1D/2D 连续时空推演任务中，长程预测误差低至 \textbf{0.09\%} 和 \textbf{3.02\%}，远超现有基准；在复杂 3D 表面流场演化中，核心物理指标误差仅 \textbf{4.06\%}。
\end{itemize}

% --- 项目二 ---
\textbf{基于图神经网络的复杂 3D 物理环境仿真与预警系统} \textit{(AI算法核心开发者)} \hfill 2024.9 -- 2025.6
\begin{itemize}
    \item \textbf{项目背景：}针对复杂物理场景下传统物理引擎（FEM）计算耗时长的痛点，开发毫秒级 3D 物理场高保真仿真预测平台。
    \item \textbf{AI 空间特征建模：}\textbf{引入 Geo-FNO} 构建可微坐标变换以处理非规则 3D 几何拓扑；\textbf{独立设计几何门控图神经网络 (GNN)}，利用表面曲率特征生成动态注意力权重，使高频形变/高曲率区域的\textbf{物理态预测误差降低 40\% 以上}。
    \item \textbf{虚实数据闭环与推演：}实现复杂 3D 拓扑与环境参数的动态解耦及高频未来态推演；将 AI 预测的理想时空基线与真实传感器回传数据进行差分比对，实现系统微小物理异常的精准捕获。
    \item \textbf{端到端系统集成：}协助团队打通了“3D态预测-多模态状态评估-大模型(LLM)专家诊断”的完整闭环，构建了具备初步自主决策能力的智能监控链路。
    \item \textbf{项目成果：}基于该项目核心算法已申请\textbf{3项发明专利}及\textbf{2项软著}，并获\textbf{第十五届中兴捧月区域优胜奖}。
\end{itemize}
