% --- 科研一：UNOP ---
\textbf{UNOP：面向长时序预测的无监督神经算子} 
\textit{(一作，IJCAI CCF-A 审稿中)} 
\hfill 2025.3 -- 至今
\begin{itemize}
    \item \textbf{背景：}现有神经PDE求解器面临标签昂贵
    （单条轨迹200+ GPU小时）、3D场景内存爆炸、
    自回归20步即发散三重瓶颈，无方法能同时解决。
    
    \item 提出积分物理约束(LIPE)替代微分残差，
    \textbf{完全消除高阶自动微分}，3D内存降至
    微分方法的\textbf{1/6}，实现\textbf{100\%无标签训练}，
    且10\%标签超越100\%全监督。
    
    \item 设计几何解耦编码器(GALA)，单一架构统一处理
    1D/2D/3D非结构化输入；曲率感知门控(GSEO)选择性
    保留高频物理结构，消融实验证明为稳定性关键（去除后
    误差$\uparrow$7倍）。
    
    \item \textbf{结果：}1D误差\textbf{0.09\%}
    （优于PINO 36$\times$），2D \textbf{3.02\%}
    （优于SP-FNO 4.5$\times$），3D Cd误差\textbf{4.06\%}，
    \textbf{50步外推仍稳定}（baseline 20步即发散）。
\end{itemize}

% --- 科研二：GeoGatedAttention ---
\textbf{GeoGatedAttention：几何感知的稀疏图Transformer} 
\textit{(发明专利已受理)} \hfill 2024.9 -- 2025.3
\begin{itemize}
    \item \textbf{背景：}3D物理场在几何突变区域
    （驻点/分离点）变化最剧烈，但标准Transformer和GAT
    的注意力均不感知边的几何结构，关键区域预测误差最大。
    
    \item 设计边级几何特征驱动的门控函数，
    生成[0,1]动态调节因子\textbf{逐边调制attention权重}，
    实现``特征相关性$\times$几何重要性''双信号融合。
    基于图稀疏结构实现\textbf{O(E)复杂度}
    （5万节点$\times$150万边）。
    
    \item 4卡DDP分布式训练，在ShapeNet-Car (889模型)
    上完成9折交叉验证。消融实验证明门控贡献最大
    （去除+17\%，优于特征拼接+15\%）。
    
    \item \textbf{结果：}压力场误差\textbf{$\downarrow$40\%}，
    Cd误差\textbf{2.69\%}，
    Spearman $\rho$=\textbf{0.97}，
    推理\textbf{20ms}。
\end{itemize}
