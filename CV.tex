\textbf{AI4Science / 神经算子方向（第一作者）}
\begin{itemize}
    \item \textbf {Physics-Constrained Unsupervised Neural Operator for Long-Horizon PDE Learning on Generalized Geometries (\textbf{一作}，IJCAI CCF-A)}
    \begin{itemize}[label=\tiny$\bullet$, leftmargin=1em]
        \item \textbf{研究痛点：} 针对无监督算子学习在长时序自回归演化中易出现的数值不稳定与预测发散难题。
        \item \textbf{核心创新：} 提出\textbf{潜空间积分物理嵌入 (LIPE)}，利用积分一致性替代传统微分约束，规避了高阶自动微分的数值震荡；设计 \textbf{GALA 适配器} 实现非结构化网格向统一潜空间的映射。
        \item \textbf{实验指标：} 在 1D/2D/3D 基准测试中，将 20 步长时序预测的误差累积显著 \textbf{降低了 60\%}，实现无标签设定下的 SOTA 精度。
    \end{itemize}
    \item \textbf {Integral-Consistent Basis-Universal Neural Operators for Diverse PDE Classes (\textbf{一作}，NeurlPS CCF-A, 在投)}
    \begin{itemize}[label=\tiny$\bullet$, leftmargin=1em]
        \item \textbf{研究痛点：} 针对传统谱算子难以在不同物理机制（\textbf{椭圆/抛物/双曲型 PDE}）间进行跨类别泛化的问题。
        \item \textbf{核心创新：} 提出\textbf{基函数通用神经算子}，通过基函数通用交互对齐机制实现了频谱幅相矩阵与权重的解耦；引入\textbf{随机积分演化}机制，动态适配物理场的瞬态与稳态变化。
        \item \textbf{实验验证：} 在统一架构下实现三类主流 PDE 的高精度求解与跨机制泛化，显著提升了模型在复杂场景下的稳健性。
    \end{itemize}
\end{itemize}

\section{\makebox[\widthof{\faBriefcase}][c]{\color{CVBlue}%
\faBriefcase}\ 国家重点研发项目}
\textbf{复杂工业三维几何下的物理场建模与数字孪生研究}
\textit{\textbf{核心成员，AI算法负责人}}
\hfill 2024.9 -- 2025.6\\[0.2em]
\textcolor{CVBlue}{\textbf{“十四五”国家重点研发计划——装备制造后市场服务协同平台研发与应用}}
\begin{itemize}
    \item \textbf{科学问题：} 针对工业场景中非结构化三维网格存在拓扑破损、尺度不一致及多源异构的问题，研究复杂几何条件下物理场建模的鲁棒表示方法。
    \item \textbf{方法设计：} 构建自动化拓扑修复与几何对齐流水线，实现 70GB 级水轮机三维模型的批量预处理；设计统一数据表示接口，使其可直接接入神经算子模型进行物理场预测。
    \item \textbf{性能突破：} 在真实工业数据上实现复杂结构场景下的稳定预测，推理时间优化至 \textbf{20ms}，支撑大规模数字孪生实时模拟。
    \item \textbf{个人职责：} 负责算法架构设计、数据建模策略制定及核心模块实现。
\end{itemize}
